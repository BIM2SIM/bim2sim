36 def run(self, elements: dict): 

37 """Enriches materials and layer sets of building elements. 

38 

39 Enrichment data in the files MaterialTemplates.json and 

40 TypeBuildingElements.json is taken from TEASER. The underlying data 

41 comes from IWU data. For more detailed information please review TEASER 

42 code documentation: 

43 https://rwth-ebc.github.io/TEASER//master/docs/index.html 

44 """ 

45 # TODO change data_source when existing for all overwritten information 

46 [element_templates, material_template] = \ 

47 yield from self.get_templates(elements) 

48 if self.playground.sim_settings.layers_and_materials is LOD.low: 

49 self.create_new_layer_sets_and_materials( 

50 elements, element_templates, material_template) 

51 

52 if self.playground.sim_settings.layers_and_materials is LOD.full: 

53 # TODO #676 

54 raise NotImplementedError("layers_and_materials full is currently" 

55 " not supported.") 

56 for layer_set in self.layer_sets_added: 

57 elements[layer_set.guid] = layer_set 

58 for layer in layer_set.layers: 

59 elements[layer.guid] = layer 

60 for material in self.template_materials.values(): 

61 elements[material.guid] = material

63 def create_new_layer_sets_and_materials( 

64 self, elements: dict, 

65 element_templates: dict, 

66 material_template: dict): 

67 """Create a new layer set including layers and materials. 

68 

69 This creates a completely new layer set, including the relevant layers 

70 and materials. Materials are only created once, even if they occur in 

71 multiple layer sets/layers. 

72 Additionally, some information on element level are overwritten with 

73 data from the templates, like inner_convection etc. 

74 """ 

75 # TODO multi building support would require to have relations 

76 # between each bim2sim element and its bim2sim Building 

77 # instance. For now we always take the first building. 

78 element_template = element_templates[ 

79 list(element_templates.keys())[0]] 

80 for template_name, ele_types in ( 

81 self.mapping_templates_bim2sim.items()): 

82 layer_set = self.create_layer_set_from_template( 

83 element_template[template_name], material_template) 

84 ele_enrichment_data = self.enrich_element_data_from_template( 

85 element_template[template_name]) 

86 elements_to_enrich = [] 

87 for ele_type in ele_types: 

88 elements_to_enrich.extend(filter_elements(elements, ele_type)) 

89 for element in elements_to_enrich: 

90 # set layer_set 

91 element.layerset = layer_set 

92 # TODO set layer_set also to not disaggregated parent, t.b.d. 

93 # if hasattr(element, "disagg_parent"): 

94 # element.disagg_parent.layerset = layer_set 

95 # overwrite element level attributes like inner_convection 

96 for att, value in ele_enrichment_data.items(): 

97 if hasattr(element, att): 

98 setattr(element, att, value) 

99 # overwrite thickness/width of element with enriched layer_set 

100 # thickness 

101 if hasattr(element, "width"): 

102 element.width = ( 

103 layer_set.thickness, AttributeDataSource.enrichment)

113 def create_layer_set_from_template(self, element_template: dict, 

114 material_template: dict) -> LayerSet: 

115 """Create layer set from template including layers and materials.""" 

116 layer_set = LayerSet() 

117 for layer_template in element_template['layer'].values(): 

118 layer = Layer() 

119 layer.thickness = layer_template['thickness'] 

120 material_name = layer_template['material']['name'] 

121 if material_name in self.template_materials: 

122 material = self.template_materials[material_name] 

123 else: 

124 material = self.create_material_from_template( 

125 material_template[material_name]) 

126 self.template_materials[material_name] = material 

127 material.parents.append(layer) 

128 layer.material = material 

129 layer.to_layerset.append(layer_set) 

130 layer_set.layers.append(layer) 

131 self.layer_sets_added.append(layer_set) 

132 return layer_set

149 def get_templates(self, elements: dict) -> object: 

150 """Get templates for elements and materials. 

151 

152 Args: 

153 elements: dict[guid: element] of bim2sim elements 

154 Returns: 

155 element_templates (dict): Holds enrichment templates for each 

156 Building with layer set information and material reference for 

157 different BPSProducts 

158 material_templates (dict): Holds information about physical 

159 attributes for each material referenced in the element_templates 

160 """ 

161 buildings = filter_elements(elements, Building) 

162 element_templates = yield from self.get_templates_for_buildings( 

163 buildings, self.playground.sim_settings) 

164 if not element_templates: 

165 self.logger.warning( 

166 "Tried to run enrichment for layers structure and materials, " 

167 "but no fitting templates were found. " 

168 "Please check your settings.") 

169 return elements, 

170 material_templates = self.get_material_templates() 

171 return element_templates, material_templates

187 resumed[material_templates[k]['name']][attr] = \

208 def _get_template_for_year(

212 element_templates = get_type_building_elements() 

213 bldg_template = {} 

214 for element_type, years_dict in element_templates.items(): 

215 if len(years_dict) == 1: 

216 template_options = years_dict[list(years_dict.keys())[0]] 

217 else: 

218 template_options = None 

219 for i, template in years_dict.items():

222 template_options = element_templates[element_type][

224 break 

225 if len(template_options) == 1:

246 bldg_template[element_type] = \
