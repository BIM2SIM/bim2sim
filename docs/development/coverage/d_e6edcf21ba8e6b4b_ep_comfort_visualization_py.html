190 def calendar_array(dates, data): 

191 i, j = zip(*[(d.day, d.month) for d in dates]) 

192 i = np.array(i) - min(i) 

193 j = np.array(j) - 1 

194 ni = max(i) + 1 

195 calendar = np.empty([ni, 12])#, dtype='S10') 

196 calendar[:] = np.nan 

197 calendar[i, j] = data 

198 return i, j, calendar
