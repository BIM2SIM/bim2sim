43 def get_location_lat_long_from_ifc(self, elements: dict) -> [float]: 

44 """ 

45 Returns the location in form of latitude and longitude based on IfcSite. 

46 

47 The location of the site and therefore the building are taken from the 

48 IfcSite in form of latitude and longitude. Latitude and Longitude each 

49 are a tuple of (degrees, minutes, seconds) and, optionally, 

50 millionths of seconds. See IfcSite Documentation for further 

51 information. 

52 Args: 

53 elements: dict with bim2sim elements 

54 

55 Returns: 

56 latitude, longitude: two float values for latitude and longitude 

57 """ 

58 site = filter_elements(elements, 'Site') 

59 if len(site) > 1: 

60 self.logger.warning( 

61 "More than one IfcSite in the provided IFC file(s). We are" 

62 "using the location of the first IfcSite found for weather " 

63 "file definition.") 

64 latitude = site[0].location_latitude 

65 longitude = site[0].location_longitude 

66 return latitude, longitude
