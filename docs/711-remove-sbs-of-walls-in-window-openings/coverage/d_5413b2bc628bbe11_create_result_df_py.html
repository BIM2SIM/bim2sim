68 def run(self, sim_results_path, bldg_names, elements): 

69 """Creates a result dataframe for BEPS simulations. 

70 

71 The created dataframe holds the data for the generic plot 

72 functionality for BEPS simulation in `bim2sim` post-processing, 

73 regardless if the simulation results come from TEASER 

74 or EnergyPlus. Therefore, the simulation results are mapped into a 

75 generic form with unified timestamps and only the results wanted, 

76 defined by the sim_settings, are exported to the dataframe. 

77 

78 Args: 

79 teaser_mat_result_paths: path to simulation result file 

80 bldg_names (list): list of all buildings 

81 elements: bim2sim elements created based on ifc data 

82 Returns: 

83 df_final: final dataframe that holds only relevant data, with 

84 generic `bim2sim` names and index in form of MM/DD-hh:mm:ss 

85 """ 

86 if not self.playground.sim_settings.dymola_simulation: 

87 self.logger.warning("Skipping task CreateResultDF as sim_setting " 

88 "'dymola_simulation' is set to False and no " 

89 "simulation was performed.") 

90 return None, 

91 if not self.playground.sim_settings.create_plots: 

92 self.logger.warning("Skipping task CreateResultDF as sim_setting " 

93 "'create_plots' is set to False and no " 

94 "DataFrame ist needed.") 

95 return None, 

96 # ToDO handle multiple buildings/ifcs #35 

97 df_finals = {} 

98 for bldg_name in bldg_names: 

99 result_path = sim_results_path / bldg_name / "teaser_results.mat" 

100 bim2sim_teaser_mapping_selected = self.select_wanted_results() 

101 # bim2sim_teaser_mapping = self.calc_indirect_result() 

102 bim2sim_teaser_mapping = self.map_zonal_results( 

103 bim2sim_teaser_mapping_selected, elements) 

104 relevant_vars = list(bim2sim_teaser_mapping.keys()) 

105 df_original = TimeSeriesData( 

106 result_path, variable_names=relevant_vars).to_df() 

107 df_final = self.format_dataframe( 

108 df_original, bim2sim_teaser_mapping, relevant_vars) 

109 df_finals[bldg_name] = df_final 

110 return df_finals,

112 def format_dataframe( 

113 self, df_original: pd.DataFrame, 

114 bim2sim_teaser_mapping: dict, 

115 relevant_vars: list) -> pd.DataFrame: 

116 """Formats the dataframe to generic bim2sim output structure. 

117 

118 This function: 

119 - adds the space GUIDs or bim2sim aggregated room names to the results 

120 - selects only the selected simulation outputs from the result 

121 - sets the index to the format MM/DD-hh:mm:ss 

122 - uses absolute values instead negative ones for cooling 

123 

124 Args: 

125 df_original: original dataframe directly taken from simulation 

126 bim2sim_teaser_mapping: dict[simulation var name: 

127 relevant_vars: list of simulation variables to have in dataframe 

128 

129 Returns: 

130 df_final: converted dataframe in `bim2sim` result structure 

131 """ 

132 

133 # rename columns based on bim2sim_teaser_mapping 

134 df_final = df_original.rename( 

135 columns=bim2sim_teaser_mapping) 

136 

137 # update index to format MM/DD-hh:mm:ss 

138 df_final = self.convert_time_index(df_final) 

139 

140 # make energy consumptions hourly instead cumulated 

141 for column in df_final.columns: 

142 if "_energy_" in column: 

143 df_final[column] = df_final[column].diff() 

144 df_final.fillna(0, limit=1, inplace=True) 

145 

146 # convert negative cooling demands and energies to absolute values 

147 df_final = df_final.abs() 

148 

149 # handle units 

150 for column in df_final: 

151 for key, unit in unit_mapping.items(): 

152 if key in column: 

153 df_final[column] = pint_pandas.PintArray( 

154 df_final[column], unit) 

155 

156 return df_final

167 def map_zonal_results(bim2sim_teaser_mapping_selected, elements: dict): 

168 """Add zone/space guids/names to mapping dict. 

169 

170 Dymola outputs the results just via a counting of zones/rooms 

171 starting with [1]. This function adds the real zone/space guids or 

172 aggregation names to the dict for easy readable results. 

173 Rooms are mapped with their space GUID, aggregated zones are mapped 

174 with their zone name. Therefore, we load the additional information 

175 like what spaces are aggregated from elements structure. 

176 

177 Args: 

178 bim2sim_teaser_mapping_selected: dict holds the mapping between 

179 simulation outputs and generic `bim2sim` output names. Only 

180 outputs selected via sim_results sim-setting are included. 

181 elements: dict[guid: element] 

182 

183 Returns: 

184 dict: A mapping between simulation results and space guids, with 

185 appropriate adjustments for aggregated zones. 

186 

187 """ 

188 bim2sim_teaser_mapping = {} 

189 space_guid_list = [] 

190 agg_tzs = filter_elements(elements, 'AggregatedThermalZone') 

191 if agg_tzs: 

192 for agg_tz in agg_tzs: 

193 space_guid_list.append(agg_tz.guid) 

194 else: 

195 tzs = filter_elements(elements, 'ThermalZone') 

196 for tz in tzs: 

197 space_guid_list.append(tz.guid) 

198 for key, value in bim2sim_teaser_mapping_selected.items(): 

199 # add entry for each room/zone 

200 if "numZones" in key: 

201 for i, space_guid in enumerate(space_guid_list, 1): 

202 new_key = key.replace("numZones", str(i)) 

203 new_value = value.replace("rooms", "rooms_" + space_guid) 

204 bim2sim_teaser_mapping[new_key] = new_value 

205 else: 

206 bim2sim_teaser_mapping[key] = value 

207 return bim2sim_teaser_mapping

211 """This converts the index of the result df to "days hh:mm:ss format""" 

212 # Convert the index to a timedelta object 

213 df.index = pd.to_timedelta(df.index, unit='s') 

214 # handle leap years 

215 if len(df.index) > 8761: 

216 year = 2020 

217 else: 

218 year = 2021 

219 # Add the specified year to the date 

220 # df.index = pd.date_range(start=f'{year}-01-01', end=f'{year+1}-01-01', freq='H') 

221 df.index = pd.to_datetime( 

222 df.index.total_seconds(), unit='s', origin=f'{year}-01-01')

228 # time steps 

229 df = df[:-1]
